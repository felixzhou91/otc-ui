useTable 列表查询 #

快速创建一个列表查询页面所需要的代码块。

列表查询页面组成：查询条件、table列表(展示数据)、分页器。

基础用法 #

<OtcTableContainer
    show-research
    @research="tableResearchFn"
    @search="tableSearchFn"
>
    <template #top>
        <OtcInput
            v-model.trim="searchForm.id"
            label="任务编号"
            size="small"
            label-width="85px"
            width="300px"
            placeholder="请输入"
        />
        <!-- 查询条件 -->
    </template>
    <template #body>
        <!-- table列表 -->
        <OtcTableList
            :loading="listLoading"
            :data="tableData"
            :columns="tableColumnList"
            height="100%"
        >
        <!-- 列表内容 slot 插槽 -->
        </OtcTableList>
    </template>

    <!-- 分页器 -->
     <template v-if="paginationTotal > 0" #bottom>
        <OtcPagination
            v-model:current-page="paginationCurrentPage"
            v-model:page-size="paginationPageSize"
            size="small"
            :total="paginationTotal"
            @change="paginationChangeFn"
        />
    </template>
</OtcTableContainer>

table列表需要展示的字段配置文件，script/data.ts

// table列表需要展示的字段配置
const tableColumnList: any[] = [
    {
        key: 'id',
        label: '消息编号',
        minWidth: '150',
        align: 'center',
        formatter: (row: any, column: any, cellValue: any) => {
            return cellValue || '-'
        }
    },
    ...
]

import { useTable } from '@common/hooks/new' // 导入hooks模块
import { fetchTableList} from '@src/api/xxx' // 导入接口api
import { tableColumnList } from './script/data'

// 根据自己需要传参或导出函数及变量
const {
    // table
    tableListRef,
    tableLoading,
    tableSearchForm,
    tableData,
    tableSearchFn,
    tableResearchFn,

    // pagination
    paginationCurrentPage,
    paginationPageSize,
    paginationTotal,
    paginationChangeFn
} = useTable(fetchTableList, {
    isImmediate: true, // 是否在初始化时调用查询接口
    queryForm: defaultData, // 默认查询条件
    currentPageKeys: 'pageNo', // 默认分页数据key
    pageSizeKeys: 'pageSize', // 默认分页数据key
    isPagination: true, // 是否开启分页
    
    // 搜索表单数据格式化
    onBeforeSearch: (data, isResearch) => {
        const params = { ...data }
        if (isResearch) {
            params.appBeginDate = workDay.value || ''
            params.appEndDate = workDay.value || ''
        }
        tableSearchForm.value = params
        return params
    },
    // 接口请求成功回调
    onSuccess: (res) => {
        const { data } = res.data || {}
        paginationTotal.value = data?.total || 0
        tableData.value = data?.list || []
    },
    // 列表查询接口请求之前的处理
    beforeCallback: async () => {
        const { data } = await queryWorkDay()
        workDay.value = data || ''
        tableSearchForm.value.appBeginDate = data
        tableSearchForm.value.appEndDate = data
    }
})

如上：一个查询列表就做好了，是不是超级easy？

有了 useTable 帮我们把大部分的工作做了，我们只需要关注业务逻辑即可，当然还得把列表tableColumnList需要展示的字段配置完成。
