useDialog 弹窗 #

快速创建弹窗 Dialog 所需要的代码块。

父组件 #

<template>
    <OtcButton size="small" link @click="dialogOpenFn(row)">编辑</OtcButton>

    <!-- 父组件调用弹窗 -->
    <DialogEdit
        :title="dialogTitle"
        v-model="dialogVisible"
        :data="dialogData"
        @close="dialogCloseFn"
        @submit="dialogSubmitFn"
    />
</template>

<script setup lang="ts">
    import DialogEdit from './DialogEdit.vue'
    import { useDialogForParent } from '@common/hooks/new' // 导入hooks模块

    // 父组件的--调用子dialog弹窗
    const { 
        // ref
        dialogTitle, // dialog标题
        dialogVisible, // dialog显示/隐藏
        dialogData, // 传递给dialog的数据

        // fn
        dialogOpenFn, // 打开弹窗
        dialogCloseFn, // 关闭弹窗
        dialogSubmitFn // 弹窗提交回调处理
    } = useDialogForParent<Record<string, any>>({
        submitCallback: () => {
            // 弹窗提交成功后刷新列表
            // tableSearchFn()
        },
        closeCallback: () => {
            // 关闭弹窗后处理
        }
    })
</script>

弹窗组件 #

<template>
    <el-dialog
        :width="600"
        :close-on-click-modal="false"
        v-bind="$attrs"
        @opened="dialogOpenFn"
    >
        <!-- dialog 内容 -->

        <template #footer>
            <OtcButton
                type="primary"
                :disabled="isLoading"
                plain
                size="small"
                @click="dialogCloseFn"
                >关闭</OtcButton
            >
            <OtcButton type="primary" :loading="isLoading" size="small" @click="dialogSubmitFn"
                >确定</OtcButton
            >
        </template>
    </el-dialog>
</template>

<script setup lang="ts">
    import { useDialogForSelf } from '@common/hooks/new' // 导入hooks模块

    const props = defineProps({
        data: {
            type: Object,
            default: () => {
                return {}
            }
        }
    })

    // 子 dialog
    const emits = defineEmits(['submit', 'close'])
    const { dialogOpenFn, dialogCloseFn, dialogSubmitFn } = useDialogForSelf({
        openCallback: async () => {
            formDataSetFn(props.data)
        },
        closeCallback: () => {
            emits('close')
        },
        submitCallback: () => {
            emits('submit')
        }
    })
</script>
